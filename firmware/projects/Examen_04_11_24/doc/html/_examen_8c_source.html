98void OperarConDistancia()
99{
100 while (true)
101 {
102 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /*la tarea espera en este punto hasta recibir la notificacion*/
103 distancia = HcSr04ReadDistanceInCentimeters();//mide la distancia del sensor
104
105 if (distancia > 500)
106 {
107 LedOn(LED_1);
108 LedOff(LED_2);
109 LedOff(LED_3);
110 }
111 else if ((distancia < 500) & (distancia > 300))
112 {
113 LedOn(LED_1);
114 LedOn(LED_2);
115 LedOff(LED_3);
116 alarmaPrecaucion();
117 escribirMensajeUART(0);
118 }
119 else if (distancia < 300)
120 {
121 LedOn(LED_1);
122 LedOn(LED_2);
123 LedOn(LED_3);
124 alarmaPeligro();
125 escribirMensajeUART(1);
126 }
127 }
128}
