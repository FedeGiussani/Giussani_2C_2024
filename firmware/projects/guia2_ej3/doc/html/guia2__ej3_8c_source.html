149void mostrarTask()
150{
151 while (true)
152 {
153 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /*la tarea espera en este punto hasta recibir la notificacion*/
154 if (on)
155 {
156 // aca se realizan las tareas de encender leds dependiendo la distancia y encender el display
157 if (distancia < 10)
158 {
159 LedsOffAll();
160 }
161 else if ((distancia > 10) & (distancia < 20))
162 {
163 LedOn(LED_1);
164 LedOff(LED_2);
165 LedOff(LED_3);
166 }
167 else if ((distancia > 20) & (distancia < 30))
168 {
169 LedOn(LED_1);
170 LedOn(LED_2);
171 LedOff(LED_3);
172 }
173 else if (distancia > 30)
174 {
175 LedOn(LED_1);
176 LedOn(LED_2);
177 LedOn(LED_3);
178 }
179
180 if (!hold)
181 {
182 LcdItsE0803Write(distancia);
183 escribirDistanciaEnPc();
184 }
185 }
186 else
187 {
188 LcdItsE0803Off();
189 LedsOffAll();
190 }
191
192 }
