137void mostrarTask()
138{
139 while (true)
140 {
141 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /*la tarea espera en este punto hasta recibir la notificacion*/
142 if (on)
143 {
144 // aca se realizan las tareas de encender leds dependiendo la distancia y encender el display
145 if (distancia < 10)
146 {
147 LedsOffAll();
148 }
149 else if ((distancia > 10) & (distancia < 20))
150 {
151 LedOn(LED_1);
152 LedOff(LED_2);
153 LedOff(LED_3);
154 }
155 else if ((distancia > 20) & (distancia < 30))
156 {
157 LedOn(LED_1);
158 LedOn(LED_2);
159 LedOff(LED_3);
160 }
161 else if (distancia > 30)
162 {
163 LedOn(LED_1);
164 LedOn(LED_2);
165 LedOn(LED_3);
166 }
167
168 if (!hold)
169 {
170 LcdItsE0803Write(distancia);
171 }
172 }
173 else
174 {
175 LcdItsE0803Off();
176 LedsOffAll();
177 }
178
179 }

211void app_main(void)
212{
213 LedsInit();
214 HcSr04Init(GPIO_3, GPIO_2);
215 LcdItsE0803Init();
216 SwitchesInit();
217
218 /* Inicialización de timer medicion */
219 timer_config_t timer_medicion = {
220 .timer = TIMER_A,
221 .period = REFRESCO_MEDICION,
222 .func_p = FuncTimerMedir,
223 .param_p = NULL};
224 TimerInit(&timer_medicion);
225
226 /* Inicialización de timer mostrar */
227 timer_config_t timer_mostrar = {
228 .timer = TIMER_B,
229 .period = REFRESCO_DISPLAY,
230 .func_p = FuncTimerMostrar,
231 .param_p = NULL};
232 TimerInit(&timer_mostrar);
233
234 /*En este caso, a partir de crear la interrupcion iniciada por el SWITCH_1 elijo que la funcion estadoTeclas
235 se va a ejecutar*/
236 SwitchActivInt(SWITCH_1, &TeclaOn, NULL);
237
238 /*En este caso, a partir de crear la interrupcion iniciada por el SWITCH_2 elijo que la funcion estadoTeclas
239 se va a ejecutar*/
240 SwitchActivInt(SWITCH_2, &TeclaHold, NULL);
241
242 /*creacion de tareas*/
243 xTaskCreate(&medirTask, "medir", 512, NULL, 5, &Medir_task_handle);
244 xTaskCreate(&mostrarTask, "mostrar", 512, NULL, 5, &Mostrar_task_handle);
245
246 /*Inicio del conteo de timers*/
247 TimerStart(timer_medicion.timer);
248 TimerStart(timer_mostrar.timer);
249}
